DELIMITER $
CREATE PROCEDURE ATTEND_CAL(IN roll int)
begin
declare attend int;
select attendance into attend from stud1 where roll_no = roll;
if attend<75 then
update stud1 set status ='ND' WHERE roll_no=roll;
else
update stud1 set status ='D' where roll_no=roll;
end if;
end;
$
